# Chakra UI v3

Accessible React component library with built-in styling, theming, and design tokens.

## Quick Start

```bash
npm i @chakra-ui/react @emotion/react

# Add pre-built component snippets (includes ColorModeButton, Provider, etc.)
npx @chakra-ui/cli snippet add
```

```tsx
// components/ui/provider.tsx (generated by CLI)
import type { ReactNode } from 'react'
import { ChakraProvider, defaultSystem } from '@chakra-ui/react'

interface ProviderProps {
  children: ReactNode
}

export function Provider({ children }: ProviderProps): JSX.Element {
  return <ChakraProvider value={defaultSystem}>{children}</ChakraProvider>
}
```

```tsx
// app/layout.tsx (Next.js App Router)
import type { ReactNode } from 'react'
import { Provider } from '@/components/ui/provider'

interface RootLayoutProps {
  children: ReactNode
}

export default function RootLayout({ children }: RootLayoutProps): JSX.Element {
  return (
    <html lang="en" suppressHydrationWarning>
      <body>
        <Provider>{children}</Provider>
      </body>
    </html>
  )
}
```

## Core Concepts

### Compound Components (v3)

Chakra v3 uses compound components with dot notation:

```tsx
import type { FC } from 'react'
import { Accordion } from '@chakra-ui/react'

interface FAQItem {
  id: string
  question: string
  answer: string
}

interface FAQProps {
  items: FAQItem[]
}

const FAQ: FC<FAQProps> = ({ items }) => {
  return (
    <Accordion.Root>
      {items.map((item) => (
        <Accordion.Item key={item.id} value={item.id}>
          <Accordion.ItemTrigger>
            {item.question}
            <Accordion.ItemIndicator />
          </Accordion.ItemTrigger>
          <Accordion.ItemContent>
            {item.answer}
          </Accordion.ItemContent>
        </Accordion.Item>
      ))}
    </Accordion.Root>
  )
}
```

### Style Props

Apply styles directly via props:

```tsx
import { Box, Flex, Text } from '@chakra-ui/react'

<Box
  bg="blue.500"
  color="white"
  p="4"
  rounded="lg"
  shadow="md"
  _hover={{ bg: 'blue.600' }}
>
  Styled box
</Box>

<Flex
  direction={{ base: 'column', md: 'row' }}
  gap="4"
  align="center"
  justify="space-between"
>
  <Text fontSize="xl" fontWeight="bold">Title</Text>
  <Text color="gray.500">Subtitle</Text>
</Flex>
```

### Common Style Props

| Prop | CSS Property | Example |
|------|--------------|---------|
| `bg` | background | `bg="blue.500"` |
| `color` | color | `color="gray.700"` |
| `p`, `px`, `py` | padding | `p="4"` `px="6"` |
| `m`, `mx`, `my` | margin | `m="auto"` `my="8"` |
| `w`, `h` | width, height | `w="full"` `h="100vh"` |
| `rounded` | border-radius | `rounded="lg"` |
| `shadow` | box-shadow | `shadow="md"` |
| `display` | display | `display="flex"` |
| `gap` | gap | `gap="4"` |

### Input and Forms

```tsx
import type { FC, FormEvent } from 'react'
import { Field, Input, Textarea, NativeSelect } from '@chakra-ui/react'

interface FormData {
  email: string
  password: string
  country: string
}

const FormExample: FC = () => {

    e.preventDefault()
    const formData = new FormData(e.currentTarget)

      <Field.Root>
```

### Dialog (Modal)

```tsx
import type { FC } from 'react'
import { useState } from 'react'
import { Button, Dialog, Portal } from '@chakra-ui/react'

interface ConfirmDialogProps {
  title: string
  message: string
  onConfirm: () => void
}

const ConfirmDialog: FC<ConfirmDialogProps> = ({ title, message, onConfirm }) => {
  const [open, setOpen] = useState<boolean>(false)

  const handleOpenChange = (details: { open: boolean }): void => {
    setOpen(details.open)
  }

  const handleConfirm = (): void => {
    onConfirm()
    setOpen(false)
  }

  return (
    <Dialog.Root open={open} onOpenChange={handleOpenChange}>
      <Dialog.Trigger asChild>
        <Button>Open Dialog</Button>
      </Dialog.Trigger>

      <Portal>
        <Dialog.Backdrop />
        <Dialog.Positioner>
          <Dialog.Content>
            <Dialog.Header>
              <Dialog.Title>{title}</Dialog.Title>
            </Dialog.Header>
            <Dialog.Body>
              {message}
            </Dialog.Body>
            <Dialog.Footer>
              <Dialog.ActionTrigger asChild>
                <Button variant="outline">Cancel</Button>
              </Dialog.ActionTrigger>
              <Button colorPalette="blue" onClick={handleConfirm}>
                Confirm
              </Button>
            </Dialog.Footer>
            <Dialog.CloseTrigger />
          </Dialog.Content>
        </Dialog.Positioner>
      </Portal>
    </Dialog.Root>
  )
}
```

### Menu

```tsx
import type { FC } from 'react'
import { Button, Menu, Portal } from '@chakra-ui/react'

interface MenuItem {
  value: string
  label: string
  color?: string
}

interface ActionMenuProps {
  items: MenuItem[]
  onSelect: (value: string) => void
}

const ActionMenu: FC<ActionMenuProps> = ({ items, onSelect }) => {
  const handleSelect = (details: { value: string }): void => {
    onSelect(details.value)
  }

  return (
    <Menu.Root onSelect={handleSelect}>
      <Menu.Trigger asChild>
        <Button variant="outline">Actions</Button>
      </Menu.Trigger>
      <Portal>
        <Menu.Positioner>
          <Menu.Content>
            {items.map((item) => (
              <Menu.Item key={item.value} value={item.value} color={item.color}>
                {item.label}
              </Menu.Item>
            ))}
          </Menu.Content>
        </Menu.Positioner>
      </Portal>
    </Menu.Root>
  )
}
```

### Tabs

```tsx
import type { FC } from 'react'

interface TabItem {
  value: string
  label: string
  content: React.ReactNode
}

interface TabsProps {
  items: TabItem[]
  defaultValue?: string
}

const TabsExample: FC<TabsProps> = ({ items, defaultValue }) => {
  return (
    <Tabs.Root defaultValue={defaultValue ?? items[0]?.value}>
      <Tabs.List>

          <Tabs.Trigger key={item.value} value={item.value}>

        <Tabs.Content key={item.value} value={item.value}>
          {item.content}
        </Tabs.Content>

  )
}
```

```tsx
import { createSystem, defaultConfig, defineConfig } from '@chakra-ui/react'

const config = defineConfig({

        brand: {
          50: { value: '#e6f2ff' },
          100: { value: '#b3d9ff' },
          200: { value: '#80bfff' },
          300: { value: '#4da6ff' },

          600: { value: '#005cb3' },
          700: { value: '#004480' },
          800: { value: '#002d4d' },
          900: { value: '#00161a' },
        },
      },
      fonts: {
        heading: { value: 'Inter, sans-serif' },
        body: { value: 'Inter, sans-serif' },
      },
    },
    semanticTokens: {
```


### Semantic Tokens

Chakra v3 provides 7 semantic tokens per color palette:

| Token | Purpose |
|-------|---------|
| `solid` | Background for solid buttons |
| `contrast` | Text on solid backgrounds |
| `fg` | Foreground/text color |
| `muted` | Muted backgrounds |
| `subtle` | Subtle backgrounds |
| `emphasized` | Hover states |
| `focusRing` | Focus ring color |

```tsx
<Box bg="brand.subtle" color="brand.fg" p="4">
  Semantic tokens adjust for light/dark mode
</Box>
```

### Dark Mode

```tsx
// components/ui/color-mode.tsx (generated by CLI snippet)
// Use the snippet: npx @chakra-ui/cli snippet add color-mode

import type { FC } from 'react'
import { ClientOnly, IconButton, Skeleton } from '@chakra-ui/react'
import { useColorMode } from '@chakra-ui/react'
import { LuMoon, LuSun } from 'react-icons/lu'

export const ColorModeButton: FC = () => {
  const { colorMode, toggleColorMode } = useColorMode()

  return (
    <ClientOnly fallback={<Skeleton boxSize="8" />}>
      <IconButton
        aria-label="Toggle color mode"
        variant="ghost"
        onClick={toggleColorMode}
      >
        {colorMode === 'light' ? <LuMoon /> : <LuSun />}
      </IconButton>
    </ClientOnly>
  )
}
```

**Note**: `ColorModeButton` is NOT a direct export from `@chakra-ui/react`. Use the CLI snippet or create your own.

## Recipes (Component Variants)

### Defining Recipes

```tsx
import { defineRecipe } from '@chakra-ui/react'
import type { RecipeVariantProps } from '@chakra-ui/react'

const buttonRecipe = defineRecipe({
  base: {
    display: 'inline-flex',
    alignItems: 'center',
    justifyContent: 'center',
    fontWeight: 'medium',
    rounded: 'lg',
    transition: 'all 0.2s',
  },
  variants: {
    visual: {
      solid: {
        bg: 'brand.500',
        color: 'white',
        _hover: { bg: 'brand.600' },
      },
      outline: {
        border: '2px solid',
        borderColor: 'brand.500',
        color: 'brand.500',
      },
    },
    size: {
      sm: { h: '8', px: '3', fontSize: 'sm' },
      md: { h: '10', px: '4', fontSize: 'md' },
      lg: { h: '12', px: '6', fontSize: 'lg' },
    },
  },
  defaultVariants: {
    visual: 'solid',
    size: 'md',
  },
})

// Type-safe variant props
type ButtonVariants = RecipeVariantProps<typeof buttonRecipe>
```

### Slot Recipes (Multi-part Components)

```tsx
import { defineSlotRecipe } from '@chakra-ui/react'
import type { SlotRecipeVariantProps } from '@chakra-ui/react'

const cardRecipe = defineSlotRecipe({
  slots: ['root', 'header', 'body', 'footer'],
  base: {
    root: {
      bg: 'white',
      rounded: 'xl',
      shadow: 'md',
      overflow: 'hidden',
    },
    header: {
      p: '4',
      borderBottom: '1px solid',
      borderColor: 'gray.200',
    },
    body: {
      p: '4',
    },
    footer: {
      p: '4',
      borderTop: '1px solid',
      borderColor: 'gray.200',
    },
  },
  variants: {
    variant: {
      elevated: {
        root: { shadow: 'xl' },
      },
      outline: {
        root: {
          shadow: 'none',
          border: '1px solid',
          borderColor: 'gray.200',
        },
      },
    },
  },
})

// Type-safe slot recipe props
type CardVariants = SlotRecipeVariantProps<typeof cardRecipe>
```

## Responsive Styles

```tsx
// Object syntax (recommended)
<Box
  fontSize={{ base: 'sm', md: 'md', lg: 'lg' }}
  p={{ base: '2', md: '4', lg: '6' }}
  display={{ base: 'block', md: 'flex' }}
>
  Responsive content
</Box>

// Array syntax (mobile-first: base, sm, md, lg, xl, 2xl)
<Box fontSize={['sm', 'md', 'lg', 'xl']}>
  Responsive text
</Box>
```

Breakpoints: `base` (0px), `sm` (480px), `md` (768px), `lg` (992px), `xl` (1280px), `2xl` (1536px)

## Form Integration (react-hook-form)

```tsx
import type { FC } from 'react'
import { useForm } from 'react-hook-form'
import { Button, Field, Input, Stack } from '@chakra-ui/react'

interface LoginFormData {
  email: string
  password: string
}

const LoginForm: FC = () => {
  const {
    register,
    handleSubmit,
    formState: { errors, isSubmitting },
  } = useForm<LoginFormData>()

  const onSubmit = async (data: LoginFormData): Promise<void> => {
    await new Promise((resolve) => setTimeout(resolve, 1000))
    console.log(data)
  }

  return (
    <form onSubmit={handleSubmit(onSubmit)}>
      <Stack gap="4">
        <Field.Root invalid={!!errors.email}>
          <Field.Label>Email</Field.Label>
          <Input
            type="email"
            {...register('email', {
              required: 'Email is required',
              pattern: {
                value: /^[A-Z0-9._%+-]+@[A-Z0-9.-]+\.[A-Z]{2,}$/i,
                message: 'Invalid email address',
              },
            })}
          />
          {errors.email && (
            <Field.ErrorText>{errors.email.message}</Field.ErrorText>
          )}
        </Field.Root>

        <Field.Root invalid={!!errors.password}>
          <Field.Label>Password</Field.Label>
          <Input
            type="password"
            {...register('password', {
              required: 'Password is required',
              minLength: {
                value: 8,
                message: 'Password must be at least 8 characters',
              },
            })}
          />
          {errors.password && (
            <Field.ErrorText>{errors.password.message}</Field.ErrorText>
          )}
        </Field.Root>

        <Button type="submit" colorPalette="blue" loading={isSubmitting}>
          Sign In
        </Button>
      </Stack>
    </form>
  )
}
```

## Best Practices

1. **Use compound components** - Prefer `Dialog.Root`, `Dialog.Content` over flat imports
2. **Leverage semantic tokens** - Use `brand.solid` instead of `brand.500` for theme flexibility
3. **Use snippets** - Run `npx @chakra-ui/cli snippet add` for pre-built patterns
4. **Responsive object syntax** - Clearer than array syntax for complex responsive styles
5. **Recipes for variants** - Define component variants in theme, not inline
6. **Type your props** - Use `RecipeVariantProps` and `SlotRecipeVariantProps` for type safety
7. **Wrap with Portal** - Always use `Portal` for overlays (Dialog, Menu, Popover) to avoid z-index issues
